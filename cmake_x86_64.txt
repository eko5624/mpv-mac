SET(CMAKE_SYSTEM_NAME Darwin)
SET(CMAKE_SYSTEM_PROCESSOR x86_64)

execute_process(COMMAND xcode-select "-print-path" OUTPUT_VARIABLE SDK_ROOT OUTPUT_STRIP_TRAILING_WHITESPACE)
set(OSX_TOOLCHAIN_ROOT "${SDK_ROOT}/Toolchains/XcodeDefault.xctoolchain")
SET(xctoolchain "${OSX_TOOLCHAIN_ROOT}/usr/bin")
SET(CMAKE_C_COMPILER "${OSX_TOOLCHAIN_ROOT}/usr/bin/clang")
SET(CMAKE_CXX_COMPILER "${OSX_TOOLCHAIN_ROOT}/usr/bin/clang++")
SET(CMAKE_RC_COMPILER "${OSX_TOOLCHAIN_ROOT}/usr/bin/windres")
SET(CMAKE_ASM_COMPILER "${OSX_TOOLCHAIN_ROOT}/usr/bin/clang")
SET(CMAKE_RANLIB "${OSX_TOOLCHAIN_ROOT}/usr/bin/ranlib")
SET(CMAKE_ASM_YASM_COMPILER yasm)

SET(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
SET(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
SET(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
